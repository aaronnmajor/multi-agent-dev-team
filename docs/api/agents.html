194    def _build_system_prompt(self, task: str) -> str:
195        if self.long_term.count() == 0:
196            return SYSTEM_PROMPT
197        memories = self.long_term.retrieve(task, top_k=3)
198        if not memories:
199            return SYSTEM_PROMPT
200        memory_block = "\n".join(f"- {m}" for m in memories)
201        return f"{SYSTEM_PROMPT}\n\n## Relevant memories from prior runs\n\n{memory_block}\n"

203    def run(self, task: str, run_id: str = "") -> AgentOutput:
204        system_prompt = self._build_system_prompt(task)
205        initial_state: AgentGraphState = {
206            "messages": [
207                SystemMessage(content=system_prompt),
208                HumanMessage(content=task),
209            ],
210            "task": task,
211            "iterations": 0,
212            "run_id": run_id,
213        }
214
215        final_state = self.graph.invoke(initial_state, {"recursion_limit": 50})
216        iterations_used = final_state.get("iterations", 0)
217        stopped_early = iterations_used >= MAX_ITERATIONS
218
219        code, result_text = self._extract_artifacts(final_state["messages"])
220        explanation, plan = self._extract_explanation_and_plan(final_state["messages"])
221
222        # Reflexion-style self-critique: run a single review pass on the
223        # produced code before returning. Skipped when there is no code to
224        # critique (e.g., the loop bailed early on a tool error) or when
225        # disabled by env var.
226        if SELF_REFLECTION_ENABLED and code.strip():
227            try:
228                with trace_span("coder", "self_reflect", run_id=run_id):
229                    reflection = self_reflect(task, code, result_text, run_id=run_id)
230                if reflection["needs_revision"] and reflection["revised_code"]:
231                    code = reflection["revised_code"]
232                if reflection["critique"]:
233                    explanation = f"{explanation}\n\n[Self-critique]\n{reflection['critique']}"
234                _log.info(
235                    "self_reflection_complete",
236                    needs_revision=reflection["needs_revision"],
237                    revised=bool(reflection["revised_code"]),
238                )
239            except Exception as exc:  # noqa: BLE001 — never let critique block the pipeline
240                _log.warning("self_reflection_failed", error=str(exc))
241
242        output = AgentOutput(
243            code=code,
244            explanation=explanation,
245            plan=plan,
246            result=result_text,
247            iterations_used=iterations_used,
248            stopped_early=stopped_early,
249        )
250
251        self.long_term.store(f"Task: {task}\nOutcome: {explanation[:500]}")
252        return output
253
254    @staticmethod
255    def _extract_artifacts(messages: list) -> tuple[str, str]:
256        code = ""
257        result = ""
258        for msg in messages:
259            tool_calls = getattr(msg, "tool_calls", None) or []
260            for call in tool_calls:
261                name = call.get("name") if isinstance(call, dict) else getattr(call, "name", "")
262                args = call.get("args") if isinstance(call, dict) else getattr(call, "args", {})
263                if name == "write_file" and isinstance(args, dict):
264                    code = args.get("content", code)
265            if getattr(msg, "type", None) == "tool" or getattr(msg, "name", None) == "exec_python":
266                result = getattr(msg, "content", result)
267        return code, result
268
269    @staticmethod
270    def _extract_explanation_and_plan(messages: list) -> tuple[str, str]:
271        assistant_texts = [
272            m.content for m in messages
273            if getattr(m, "type", None) == "ai" and getattr(m, "content", None)
274        ]
275        if not assistant_texts:
276            return "", ""
277        plan = assistant_texts[0]
278        explanation = assistant_texts[-1]
279        return explanation, plan

315def coder_node(state: ProjectState) -> dict[str, Any]:
316    """LangGraph node: process the current task and route to QA for review.
317
318    Reads `tasks[current_task_index]`, runs the single-agent graph on it,
319    writes the resulting artifact back, and routes to "qa". Task index is
320    advanced by the QA node on pass (or on retry exhaustion), not here.
321    """
322    tasks = state.get("tasks", [])
323    idx = state.get("current_task_index", 0)
324
325    if idx >= len(tasks):
326        return {"routing": "done"}
327
328    task = tasks[idx]
329    spec = state.get("tech_spec", "")
330    instruction = _task_instruction(task, spec)
331
332    # If this is a retry, append the previous QA feedback so the coder can fix the issues.
333    feedback = _format_feedback_from_latest_review(state, task["task_id"])
334    if feedback:
335        instruction = f"{instruction}\n\n---\n{feedback}"
336
337    run_id = state.get("run_id", "")
338    with trace_span("coder", "coder_task", run_id, task_id=task["task_id"]):
339        agent = CoderAgent()
340        output = agent.run(instruction, run_id=run_id)
341
342    artifact: CodingArtifact = {
343        "task_id":     task["task_id"],
344        "file":        task["file"],
345        "content":     output.code,
346        "exec_result": output.result,
347    }
348
349    _log.info(
350        "coder_task_complete",
351        run_id=run_id,
352        task_id=task["task_id"],
353        code_chars=len(output.code),
354        iterations=output.iterations_used,
355    )
356    return {
357        "artifacts": [artifact],
358        "routing":   "qa",
359    }

199def pm_node(state: ProjectState) -> dict[str, Any]:
200    """LangGraph node: run the PM agent on the current state."""
201    requirement = state.get("user_requirement", "").strip()
202    run_id = state.get("run_id", "")
203
204    if not requirement:
205        _log.error("pm_missing_requirement", run_id=run_id)
206        return {"routing": "error", "error": "Missing user_requirement in state"}
207
208    with trace_span("pm", "pm_node", run_id):
209        client = _client()
210        if _debate_enabled():
211            with trace_span("pm", "spec_debate", run_id):
212                debate = propose_spec_with_debate(
213                    requirement, model=MODEL, client=client, run_id=run_id,
214                )
215            spec = debate["spec"]
216        else:
217            with trace_span("pm", "build_tech_spec", run_id):
218                spec = build_tech_spec(requirement, client=client, run_id=run_id)
219        with trace_span("pm", "decompose_into_tasks", run_id):
220            tasks = decompose_into_tasks(spec, client=client, run_id=run_id)
221        if len(tasks) > MAX_TASKS_PER_REQUIREMENT:
222            with trace_span("pm", "consolidate_tasks", run_id, original_count=len(tasks)):
223                tasks = consolidate_tasks(tasks, client=client, run_id=run_id)
224            _log.info("pm_consolidated", run_id=run_id, final_count=len(tasks))
225
226    if not tasks:
227        _log.error("pm_zero_tasks", run_id=run_id)
228        return {
229            "tech_spec": spec,
230            "routing": "error",
231            "error": "PM agent produced zero coding tasks from the spec",
232        }
233
234    _log.info("pm_complete", run_id=run_id, task_count=len(tasks))
235    return {
236        "tech_spec": spec,
237        "tasks": tasks,
238        "current_task_index": 0,
239        "routing": "coder",
240    }

def qa_node(state: orchestration.state.ProjectState) -> dict[str, typing.Any]: 
123def qa_node(state: ProjectState) -> dict[str, Any]:
124    """LangGraph node: review the most recent artifact and route accordingly.
125
126    Reads the latest artifact (from the Coder's previous turn) and the task it
127    corresponds to. Routes back to "coder" if the review failed AND we haven't
128    exhausted MAX_RETRIES_PER_TASK; otherwise advances the task index and
129    routes to "coder" (more tasks) or "done" (queue drained).
130    """
131    artifacts = state.get("artifacts", [])
132    tasks = state.get("tasks", [])
133    current_idx = state.get("current_task_index", 0)
134    retry_count = state.get("retry_count", 0)
135
136    if not artifacts:
137        return {"routing": "done"}
138
139    latest = artifacts[-1]
140    task_id = latest.get("task_id", "")
141    task = next((t for t in tasks if t.get("task_id") == task_id), None)
142    if task is None:
143        return {"routing": "done"}
144
145    run_id = state.get("run_id", "")
146    with trace_span("qa", "qa_review", run_id, task_id=task_id):
147        review = review_artifact(task, latest, run_id=run_id)
148    _log.info(
149        "qa_review_complete",
150        run_id=run_id,
151        task_id=task_id,
152        passed=review["passed"],
153        issues=len(review.get("issues", [])),
154    )
155
156    if not review["passed"] and retry_count < MAX_RETRIES_PER_TASK:
157        # Fail and retry: keep the same task; the coder will redo it using the review feedback.
158        return {
159            "reviews":     [review],
160            "retry_count": retry_count + 1,
161            "routing":     "coder",
162        }
163
164    # Pass OR out of retries: advance to the next task.
165    next_idx = current_idx + 1
166    done = next_idx >= len(tasks)
167    return {
168        "reviews":            [review],
169        "retry_count":        0,
170        "current_task_index": next_idx,
171        "routing":            "done" if done else "coder",
172    }

 72) -> QAReview:
 73    """Run a single QA review on one artifact. Returns a QAReview dict."""
 74    client = client or _client()
 75
 76    user_message = (
 77        f"Task:\n"
 78        f"  ID: {task.get('task_id', '')}\n"
 79        f"  Title: {task.get('title', '')}\n"
 80        f"  Description: {task.get('description', '')}\n"
 81        f"  Acceptance criteria: {task.get('acceptance_criteria', '')}\n"
 82        f"  Target file: {task.get('file', '')}\n\n"
 83        f"Artifact:\n"
 84        f"  File: {artifact.get('file', '')}\n"
 85        f"  Content:\n```python\n{artifact.get('content', '')}\n```\n"
 86        f"  Execution result:\n{artifact.get('exec_result', '')}\n"
 87    )
 88
 89    response = client.chat.completions.create(
 90        model=MODEL,
 91        messages=[
 92            {"role": "system", "content": REVIEW_PROMPT},
 93            {"role": "user", "content": user_message},
 94        ],
 95    )
 96    record_usage_from_response(run_id, "qa", MODEL, response)
 97    raw = (response.choices[0].message.content or "").strip()
 98    if raw.startswith("```"):
 99        raw = raw.split("\n", 1)[1] if "\n" in raw else raw
100        raw = raw.rsplit("```", 1)[0].strip()
101
102    try:
103        parsed = json.loads(raw)
104    except json.JSONDecodeError:
105        # Default to pass on parse failure so we don't block on a bad response.
106        return {
107            "task_id":     task.get("task_id", ""),
108            "passed":      True,
109            "issues":      [],
110            "suggestions": [],
111            "summary":     "QA review produced malformed JSON; defaulting to pass.",
112        }
113
114    return {
115        "task_id":     task.get("task_id", ""),
116        "passed":      bool(parsed.get("passed", True)),
117        "issues":      list(parsed.get("issues", [])),
118        "suggestions": list(parsed.get("suggestions", [])),
119        "summary":     str(parsed.get("summary", "")),
120    }
